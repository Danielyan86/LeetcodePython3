class Solution:
    def calculate(self, s: str) -> int:
        res, cur, num = 0, 0, 0
        sign='+'

        for char in s+'+':
            if char == ' ': continue
            if char.isdigit():
                num = num*10 + int(char)
                continue  
            if sign == "+":
                res = res + cur
                cur = num
            elif sign == "-":
                res = res + cur
                cur = 0-num
            elif sign == "*":
                cur = cur*num
            elif sign == "/":
                cur = int(cur/num)
            sign = char
            num = 0


        return res+ cur
